💑 SQL: Vínculo de pareja (ejecutar una sola vez) 
-- 1) Tabla de vínculo confirmado
CREATE TABLE parejas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_a uuid REFERENCES auth.users NOT NULL,
  user_b uuid REFERENCES auth.users NOT NULL,
  email_a text,
  email_b text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE parejas ENABLE ROW LEVEL SECURITY;
CREATE POLICY "parejas_select" ON parejas FOR SELECT
  USING (auth.uid() = user_a OR auth.uid() = user_b);
CREATE POLICY "parejas_insert" ON parejas FOR INSERT
  WITH CHECK (
    auth.uid() = user_b
    AND EXISTS (SELECT 1 FROM pareja_solicitudes s
      WHERE s.from_user = parejas.user_a
        AND lower(s.to_email) = lower(auth.jwt()->>'email')
        AND s.estado = 'pendiente')
  );
CREATE POLICY "parejas_delete" ON parejas FOR DELETE
  USING (auth.uid() = user_a OR auth.uid() = user_b);

-- 2) Tabla de invitaciones por email
CREATE TABLE pareja_solicitudes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  from_user uuid REFERENCES auth.users NOT NULL,
  from_email text,
  to_email text NOT NULL,
  estado text DEFAULT 'pendiente',
  created_at timestamptz DEFAULT now()
);
ALTER TABLE pareja_solicitudes ENABLE ROW LEVEL SECURITY;
CREATE POLICY "solicitudes_select" ON pareja_solicitudes FOR SELECT
  USING (auth.uid() = from_user OR lower(to_email) = lower(auth.jwt()->>'email'));
CREATE POLICY "solicitudes_insert" ON pareja_solicitudes FOR INSERT
  WITH CHECK (auth.uid() = from_user);
CREATE POLICY "solicitudes_update" ON pareja_solicitudes FOR UPDATE
  USING (auth.uid() = from_user OR lower(to_email) = lower(auth.jwt()->>'email'));
CREATE POLICY "solicitudes_delete" ON pareja_solicitudes FOR DELETE
  USING (auth.uid() = from_user);

-- 3) Reemplazar políticas de gastos e ingresos: de "cualquier usuario" a "solo tu pareja vinculada"
DO $$
DECLARE pol record;
BEGIN
  FOR pol IN SELECT policyname FROM pg_policies WHERE tablename = 'gastos' LOOP
    EXECUTE format('DROP POLICY %I ON gastos', pol.policyname);
  END LOOP;
  FOR pol IN SELECT policyname FROM pg_policies WHERE tablename = 'ingresos' LOOP
    EXECUTE format('DROP POLICY %I ON ingresos', pol.policyname);
  END LOOP;
END $$;

CREATE POLICY "gastos_select" ON gastos FOR SELECT USING (
  auth.uid() = user_id
  OR (es_compartido = true AND EXISTS (
    SELECT 1 FROM parejas p WHERE (p.user_a = auth.uid() AND p.user_b = gastos.user_id)
                                OR (p.user_b = auth.uid() AND p.user_a = gastos.user_id)))
);
CREATE POLICY "gastos_insert" ON gastos FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "gastos_update" ON gastos FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "gastos_delete" ON gastos FOR DELETE USING (auth.uid() = user_id);

CREATE POLICY "ingresos_select" ON ingresos FOR SELECT USING (
  auth.uid() = user_id
  OR EXISTS (
    SELECT 1 FROM parejas p WHERE (p.user_a = auth.uid() AND p.user_b = ingresos.user_id)
                                OR (p.user_b = auth.uid() AND p.user_a = ingresos.user_id))
);
CREATE POLICY "ingresos_insert" ON ingresos FOR INSERT WITH CHECK (auth.uid() = user_id);
CREATE POLICY "ingresos_update" ON ingresos FOR UPDATE USING (auth.uid() = user_id);
CREATE POLICY "ingresos_delete" ON ingresos FOR DELETE USING (auth.uid() = user_id);
